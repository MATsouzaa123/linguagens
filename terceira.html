    // C++ code  
    int POT1 = 0;
    int POT2 = 0;
    int RES = 0;
    void setup()
    {
      pinMode(12, OUTPUT);
      pinMode(A0, INPUT);
      Serial.begin(9600);
      pinMode(A1, INPUT);
      POT1 = 0;
      POT2 += 0;
      RES = 0;
      digitalWrite(12, LOW);
    }
    void loop()
    {
      POT1 += analogRead(A0);
      Serial.println("pot 1-------------------------------------------------------");
      Serial.println(POT1);
      POT2 += analogRead(A1);
      Serial.println("pot 2-------------------------------------------------------------------");
      Serial.println(POT2);
      if (POT1 > POT2) {
        RES += (POT1 + POT2);
        Serial.println("res-------------------------------------------------------------------------");
        Serial.println(RES);
        digitalWrite(12, HIGH);
      }
      delay(1); // Wait for 1 millisecond(s)
    }
